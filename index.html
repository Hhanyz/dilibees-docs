// app/Enums/CandidateStatus.php
enum CandidateStatus: string {
    case REGISTERED = 'registered';
    case CV_COMPLETE = 'cv_complete';
    case WAITING_MATCH = 'waiting_match';
    case SUITABLE = 'suitable';         // vhodný pro poptávku
    case IN_PROGRESS = 'in_progress';   // v řešení
    case TRIAL = 'trial';               // na zkoušce
    case HIRED = 'hired';               // k nástupu
    case ACTIVE = 'active';             // aktivní pracovník
    case REJECTED = 'rejected';         // nedohodli se
    case TERMINATED = 'terminated';     // ukončení

    public function canTransitionTo(self $new): bool {
        return match($this) {
            self::REGISTERED => in_array($new, [self::CV_COMPLETE]),
            self::CV_COMPLETE => in_array($new, [self::WAITING_MATCH]),
            self::WAITING_MATCH => in_array($new, [self::SUITABLE]),
            self::SUITABLE => in_array($new, [self::IN_PROGRESS]),
            self::IN_PROGRESS => in_array($new, [self::TRIAL, self::REJECTED]),
            self::TRIAL => in_array($new, [self::HIRED, self::REJECTED]),
            self::HIRED => in_array($new, [self::ACTIVE]),
            self::ACTIVE => in_array($new, [self::TERMINATED]),
            self::REJECTED => in_array($new, [self::WAITING_MATCH]),
            self::TERMINATED => in_array($new, [self::WAITING_MATCH]),
        };
    }
}

Laravel implementace — Notifikační třídy
// php artisan make:notification NewMessageNotification
class NewMessageNotification extends Notification {
    use Queueable;

    public function __construct(
        public Message $message,
        public Conversation $conversation
    ) {}

    public function via($notifiable): array {
        return ['database', 'broadcast']; // mail volitelně
    }

    public function toDatabase($notifiable): array {
        return [
            'type' => 'new_message',
            'title' => 'Nová zpráva',
            'body' => Str::limit($this->message->body, 80),
            'conversation_id' => $this->conversation->id,
            'sender_id' => $this->message->user_id,
            'sender_name' => $this->message->user->full_name,
            'url' => route('messages.show', $this->conversation),
        ];
    }

    public function toBroadcast($notifiable): BroadcastMessage {
        return new BroadcastMessage($this->toDatabase($notifiable));
    }
}

Database migrace — conversations + messages
Schema::create('conversations', function (Blueprint $table) {
    $table->id();
    $table->string('subject')->nullable(); // null u chatu, vyplněno u agent e-mailu
    $table->enum('type', ['chat', 'email'])->default('chat');
    $table->timestamp('last_message_at')->nullable()->index();
    $table->timestamps();
    $table->softDeletes();
});

Schema::create('conversation_user', function (Blueprint $table) {
    $table->id();
    $table->foreignId('conversation_id')->constrained()->cascadeOnDelete();
    $table->foreignId('user_id')->constrained()->cascadeOnDelete();
    $table->timestamp('last_read_at')->nullable();
    $table->boolean('is_archived')->default(false);
    $table->unique(['conversation_id', 'user_id']);
});

Schema::create('messages', function (Blueprint $table) {
    $table->id();
    $table->foreignId('conversation_id')->constrained()->cascadeOnDelete();
    $table->foreignId('user_id')->constrained(); // odesílatel
    $table->text('body');
    $table->nullableMorphs('attachable'); // polymorphic: User(kandidát), Document, atd.
    $table->timestamps();
    $table->softDeletes();
    $table->index(['conversation_id', 'created_at']);
});

Laravel implementace

Sdílený model Conversation + Message (viz globální messaging). Konverzace typu chat.

// Automatické vytvoření konverzace hunter-admin při první zprávě
public function getOrCreateHunterConversation(User $hunter): Conversation
{
    $admin = User::where('role', 'admin')->first(); // nebo admin pool
    $existing = $hunter->conversations()
        ->whereHas('users', fn($q) => $q->where('role', 'admin'))
        ->where('type', 'chat')
        ->first();
    if ($existing) return $existing;

    $conv = Conversation::create(['type' => 'chat']);
    $conv->users()->attach([$hunter->id, $admin->id]);
    return $conv;
}

Více adminů v systému — kdo odpovídá hunterovi? Řešení: Konverzace je s „admin pool". Jakýkoliv admin může odpovědět. Zobrazit jméno konkrétního admina u zprávy.

Hunter pošle zprávu když je admin offline. Zpráva se uloží, admin vidí při přihlášení. Email notifikace adminovi (volitelně, nastavitelné).

public function data(Request $request) {
    $hunterId = $request->user()->id;

    // Line chart data — posledních 12 měsíců
    $monthlyRegistrations = User::where('referred_by', $hunterId)
        ->where('created_at', '>=', now()->subMonths(12))
        ->selectRaw("DATE_FORMAT(created_at, '%Y-%m') as month, COUNT(*) as count")
        ->groupBy('month')
        ->orderBy('month')
        ->pluck('count', 'month');

    // Donut chart — status breakdown
    $statusBreakdown = User::where('referred_by', $hunterId)
        ->selectRaw("candidate_status, COUNT(*) as count")
        ->groupBy('candidate_status')
        ->pluck('count', 'candidate_status');

        'monthly' => $monthlyRegistrations,
        'statuses' => $statusBreakdown,
        'total' => User::where('referred_by', $hunterId)->count(),
        'active' => $statusBreakdown->get('active', 0),
        'success_rate' => round(($statusBreakdown->get('active', 0) / max($statusBreakdown->sum(), 1)) * 100, 1),

}

Automatický matching hodin
// Observer: HourEntryObserver::updated()
public function updated(HourEntry $entry) {
    if ($entry->hours_worker !== null && $entry->hours_agent !== null) {
        $tolerance = 2; // povolený rozdíl 2 hodiny
        if (abs($entry->hours_worker - $entry->hours_agent) <= $tolerance) {
            $entry->update(['match_status' => 'matched']);
        } else {
            $entry->update(['match_status' => 'disputed']);
            // Notifikace všem třem stranám
            Notification::send(
                [$entry->user, $entry->agent, ...User::admins()->get()],
                new HoursDisputeNotification($entry)
            );
        }
    }
}
